\subsection{Tables} % (fold)
\label{sub:tables}

\textbf{A cool table:} \par

\begin{center}
	\begin{tabular}[H]{l*{3}{c}}
		\hline
		A thing & A value & Another value \\
		\hline
		Thing 1 & 23 & false \\
		Thing 2 & 857 & true \\
		\hline
	\end{tabular}
\end{center}

\par

\textbf{Another cool table:}\par

\begin{center}
	\begin{tabular}[H]{l | c | r}
		\hline
		A thing & A value & Another value \\
		\hline
		Thing 1 & 23 & false \\
		Thing 2 & 857 & true \\
		\hline
	\end{tabular}
\end{center}

		\end{itemize}
	\item Item 2
		\begin{itemize}
			\item A
				\begin{itemize}
					\item B
				\end{itemize}
		\end{itemize}

\begin{enumerate}

		\begin{enumerate}

			\item B
		\end{enumerate}
